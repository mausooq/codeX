Check Even/Odd


#include <stdio.h>

    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if(num % 2 == 0)
        printf("Even");
    else
        printf("Odd");
    return 0;
}

Factorial Program


#include <stdio.h>
int main() {
    int n = 5, fact = 1;
    for(int i = 1; i <= n; i++)
        fact *= i;
    printf("Factorial = %d", fact);
    return 0;
}

Check Prime Number


#include <stdio.h>
int main() {
    int n = 7, isPrime = 1;
    for(int i = 2; i < n; i++) {
        if(n % i == 0) {
            isPrime = 0;
            break;
        }
    }
    printf(isPrime ? "Prime" : "Not Prime");
    return 0;
}

Fibonacci Series


#include <stdio.h>
int main() {
    int n1 = 0, n2 = 1, n3;
    printf("%d %d ", n1, n2);
    for(int i = 2; i < 10; i++) {
        n3 = n1 + n2;
        printf("%d ", n3);
        n1 = n2;
        n2 = n3;
    }
    return 0;
}

Swap Numbers


#include <stdio.h>
int main() {
    int a = 5, b = 10, temp;
    temp = a;
    a = b;
    b = temp;
    printf("a=%d b=%d", a, b);
    return 0;
}

Reverse Number


#include <stdio.h>
int main() {
    int n = 123, rev = 0;
    while(n > 0) {
        rev = rev*10 + n%10;
        n /= 10;
    }
    printf("Reversed: %d", rev);
    return 0;
}
