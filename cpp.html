Object-Oriented Programming Example


#include <iostream>
using namespace std;

class Student {
private:
    string name;
    int age;
    float gpa;

public:
    Student(string n, int a, float g) {
        name = n;
        age = a;
        gpa = g;
    }

    void display() {
        cout << "Name: " << name << endl;
        cout << "Age: " << age << endl;
        cout << "GPA: " << gpa << endl;
    }
};

int main() {
    Student student("John Doe", 20, 3.8);
    student.display();
    return 0;
}

Template Function


#include <iostream>
using namespace std;

template <typename T>
T findMax(T a, T b) {
    return (a > b) ? a : b;
}

int main() {
    cout << "Max of 10 and 20: " << findMax(10, 20) << endl;
    cout << "Max of 3.14 and 2.78: " << findMax(3.14, 2.78) << endl;
    cout << "Max of 'a' and 'z': " << findMax('a', 'z') << endl;
    return 0;
}

Vector Operations


#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> vec = {1, 2, 3, 4, 5};
    int sum = 0;
    for (int i = 0; i < vec.size(); i++) {
        sum += vec[i];
    }
    cout << "Sum of vector elements: " << sum << endl;
    return 0;
}
